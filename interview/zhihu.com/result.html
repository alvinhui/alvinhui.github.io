第二题：因为1秒之后调用函数的默认的是window对象，所以sayHello里的this指针为window，只需要绑定到foo即可。

foo = function(){
    this.myName = "Foo function.";
}
foo.prototype.sayHello = function(){
    alert(this.myName);
}
foo.prototype.bar = function(){
    /*或setTimeout(this.sayHello.call(this), 1000);*/
    setTimeout(this.sayHello.bind(this), 1000);
}
var f = new foo;
f.bar();
